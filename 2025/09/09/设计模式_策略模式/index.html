public interface SearchExecutor {

List dense, List sparse,

private <T> Set<T> searchWithStrategy(
        SseEmitterUTF8 emitter,
        List<String> questions,
        List<Integer> kbIds,
        RagStrategyEnum strategy,
        String collectionPrefix,
        Function<RagStrategyEnum, SearchExecutor<T>> executorFactory,
        double threshold
) {
    Set<T> result = new HashSet<>();
    sendSse(emitter, "searching");

    try {
        List<BaseVector> dense = null, sparse = null;
        if (strategy != RagStrategyEnum.KEYWORD) {
            Map<String, Object> vectors = getVector(questions);
            dense = (List<BaseVector>) vectors.get("dense");
            sparse = (List<BaseVector>) vectors.get("sparse");
        }

        SearchExecutor<T> executor = executorFactory.apply(strategy);

        for (Integer kbId : kbIds) {
            try {
                result.addAll(executor.execute(
                        collectionPrefix + kbId, questions, dense, sparse, threshold, emitter
                ));
            } catch (Exception ex) {
                log.warn("{} æŸ¥è¯¢å¤±è´¥ï¼Œè·³è¿‡ kbId={}ï¼ŒåŸå› ï¼š{}", strategy, kbId, ex.getMessage(), ex);
            }
        }
    } finally {
        sendSse(emitter, "searching_done");
    }
    return result;
}

public Set<QaRerankResponse> getRelatedQaResult(
        SseEmitterUTF8 emitter, List<String> questions,
        List<Integer> kbIds, RagStrategyEnum strategy, double threshold) {
    sendSse(emitter, "searching_in_qa");
    Set<QaRerankResponse> qaRerankResponses = searchWithStrategy(emitter, questions, kbIds, strategy,
            milvusConfig.getQa().getCollection(),
            s -> switch (s) {

                        (c, q, d, sp, th, e) -> milvusService.searchQaByStrategy(c, q, null, null, RagStrategyEnum.KEYWORD, e, th);
                case VECTOR ->
                        (c, q, d, sp, th, e) -> milvusService.searchQaByStrategy(c, q, d, sp, RagStrategyEnum.VECTOR, e, th);
                case HYBRID ->
                        (c, q, d, sp, th, e) -> milvusService.searchQaByStrategy(c, q, d, sp, RagStrategyEnum.HYBRID, e, th);
            },
            threshold
    );

}
